% --- Better code-block typography -------------------------------------------
% Loaded here too. `-H preamble.tex` lands in `$for(header-includes)$`,
% which precedes `$highlighting-macros$` in pandoc's template — so this
% \DefineVerbatimEnvironment runs early enough to be authoritative.
% `commandchars=\\\{\}` is required for pandoc's syntax-highlighting
% commands (\ExtensionTok, \NormalTok, \AttributeTok, etc.) to expand
% inside the Verbatim environment; without it they render as literal text.
\usepackage{fvextra}
\DefineVerbatimEnvironment{Highlighting}{Verbatim}{
  commandchars=\\\{\},
  breaklines,
  breakanywhere,
  fontsize=\footnotesize,
}

% --- Inline-code line-breaking (\seqsplit for monospace) --------------------
% Pandoc renders inline `code` as \texttt{...}. Long monospace strings
% (mk1q…, xpub6…, paths, long identifiers) inside \texttt have no break-
% opportunities and overflow the right margin in body paragraphs, tables,
% and code-block tokens. Use the seqsplit package (battle-tested
% \futurelet-based char-splitter) and create a \brktt wrapper. We do
% NOT redefine \texttt globally — that causes infinite-loop hangs in
% moving-argument contexts (TOC entries, section headings, .aux/.out
% re-reads with pandoc-emitted \texttt inside \section{}).
%
% Instead, the Lua filter at pandoc/filters/wrap-long-code.lua walks
% inline `Code` AST nodes only (NOT Header content), and when their
% text contains a non-space run >=12 chars, emits \brktt{...} via a
% RawInline. Section titles, captions, and bookmarks keep using vanilla
% \texttt and are never touched.
\usepackage{seqsplit}
\newcommand{\brktt}[1]{\texttt{\seqsplit{#1}}}

% xurl: aggressive URL line-breaking. Default `url` package only breaks
% at limited URL positions; xurl breaks at any character, which is what
% we need for long GitHub release URLs that overflow the right margin.
\usepackage{xurl}

% Shrink longtable contents slightly. Pandoc emits all markdown tables
% as longtable; the @{}ll@{} default column spec has no fixed widths
% and overflows when cells contain inline-code + descriptive text.
% \small (≈10pt body at 11pt base) gives ~10% extra column room without
% hurting readability.
\AtBeginEnvironment{longtable}{\small}

% --- TOC page-number + section-number gutter widening -------------------
% Two distinct issues:
%
% 1. book.cls's default \@pnumwidth (1.55em) is too narrow for 3-digit
%    page numbers — every TOC line at page >= 100 emits a ~4pt overfull
%    hbox warning. Widening to 2.5em fits 3-digit numbers cleanly.
%
% 2. book.cls's \@dottedtocline numwidth for sections (2.3em) is too
%    narrow for 2-digit minor section numbers like "29.10" — emits a
%    ~6pt overfull hbox per affected TOC row. Widening section/subsection/
%    subsubsection numwidth slightly handles X.10+ cleanly.
\makeatletter
\renewcommand{\@pnumwidth}{2.5em}
\renewcommand{\@tocrmarg}{3em}
\renewcommand{\l@section}{\@dottedtocline{1}{1.5em}{2.8em}}
\renewcommand{\l@subsection}{\@dottedtocline{2}{4.0em}{3.8em}}
\renewcommand{\l@subsubsection}{\@dottedtocline{3}{7.5em}{4.6em}}
\makeatother

% --- Suppress sub-1pt overfull-hbox warnings -----------------------------
% TeX flags every horizontal overflow above \hfuzz (default: 0.1pt).
% After the TOC widening above, residual TOC overruns are 0.6-1pt —
% invisible in the rendered PDF but ~100 log warnings. Raising \hfuzz
% to 1pt suppresses cosmetic noise without affecting any layout that
% a reader would notice. Real overflows (>= 1pt) still warn.
\setlength{\hfuzz}{1pt}

% --- Title-page constellation graphic (m-format constellation) -----------------------
\usepackage{tikz}
\usetikzlibrary{shapes.geometric}
\usepackage{etoolbox}

\newcommand{\mformatconstellation}{%
  \begin{tikzpicture}[scale=1.2]
    \fill[black!92] (-2.7,-2.2) rectangle (2.7,2.2);
    % faint background stars
    \foreach \x/\y in {-2.3/1.5, -1.7/-1.7, -0.6/0.9, 0.4/-1.9, 1.9/1.7, 2.3/-0.4, 1.0/1.2, -2.0/0.3, 0.7/-1.0, -1.1/1.8, 2.1/0.6} {
      \fill[white] (\x,\y) circle (0.5pt);
    }
    % three labeled bright stars
    \node[star, star points=5, star point ratio=2.25, fill=white, draw=white, inner sep=2.5pt] (ms) at (0.4, 1.5) {};
    \node[right=3pt, white, font=\small] at (ms) {ms1};
    \node[star, star points=5, star point ratio=2.25, fill=white, draw=white, inner sep=2.2pt] (mk) at (-1.6, 0.1) {};
    \node[left=3pt, white, font=\small] at (mk) {mk1};
    \node[star, star points=5, star point ratio=2.25, fill=white, draw=white, inner sep=2.2pt] (md) at (1.5, -0.7) {};
    \node[right=3pt, white, font=\small] at (md) {md1};
    % connecting strokes
    \draw[white!55, very thin] (ms) -- (mk);
    \draw[white!55, very thin] (ms) -- (md);
    \draw[white!55, very thin] (mk) -- (md);
  \end{tikzpicture}%
}

% Inject the constellation below the title block (i.e., at the end of
% pandoc's titlepage environment, after \@title / \@author / \@date).
\AtEndEnvironment{titlepage}{%
  \vspace{2em}%
  \begin{center}\mformatconstellation\end{center}%
}
