% =========================================================================
% Page 1 — top disc (rotating overlay).
% =========================================================================
\begin{center}
% Top disc: scaled to fit letter page. Max raw extent ≈ 4.9" (license footer)
% vs. ~3.75" usable radius after 0.5" margins; scale ≈ 0.75 keeps margin-safe.
\begin{tikzpicture}[rotate=-90, scale=0.75]
  % 32 input-symbol labels at radius 3.5", 11.25° per segment, 10pt.
  \foreach \sym [count=\i from 0] in \WheelAlphabet {%
    \node at ({\i*11.25}:3.5in) [font=\sffamily\fontsize{10}{12}\selectfont,
      rotate={90 - \i*11.25}] {\sym};%
    % Segment divider (thin radial tick).
    \draw[line width=0.2pt] ({(\i+0.5)*11.25}:3.4in)
      -- ({(\i+0.5)*11.25}:3.6in);%
  }
  % Outer ring of input symbols.
  \draw[line width=0.4pt] (0,0) circle (3.4in);
  \draw[line width=0.4pt] (0,0) circle (3.6in);

  % 32 viewing-window outlined rectangles between radius 1.5" and 2.5",
  % one per 11.25° segment. The user scissors these out after printing.
  \foreach \i in {0,...,31} {%
    \draw[line width=0.3pt]
      ({(\i-0.4)*11.25}:1.5in) -- ({(\i-0.4)*11.25}:2.5in)
      -- ({(\i+0.4)*11.25}:2.5in) -- ({(\i+0.4)*11.25}:1.5in) -- cycle;%
  }

  % Index pointer: small filled triangle at 12 o'clock pointing inward.
  % After the rotate=-90 frame, "12 o'clock" is angle 0.
  \fill (0:3.65in) -- (-2.5:3.85in) -- (2.5:3.85in) -- cycle;

  % Format-specific endpoint mark.
  \volvelleendpoint{\WheelEndpointAngle}{\WheelFormat\ endpoint}

  % License footer.
  \node at (0,-4.9in) [font=\sffamily\fontsize{7}{8}\selectfont, rotate=90]
    {\copyright\ 2026 Brian Goss --- MIT License \quad \WheelFormat\ top disc};
\end{tikzpicture}
\end{center}
